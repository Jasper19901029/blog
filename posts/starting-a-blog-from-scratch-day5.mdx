
`css`的`>`是代表選擇包含在內的樣式，例如上面的`html > body > div > p > code`，是選擇包含在`html`、`body`、`div`、`p`內`code`的樣式。

```typescript:app/mdx/markdown-component.tsx showLineNumbers {7-20}
// app/mdx/markdown-component.tsx
import { MDXComponents } from "mdx/types";
import Link from "next/link";
import { AnchorHTMLAttributes } from "react";

export const mdxComponents: MDXComponents = {
  a: ({
    children,
    href,
  }: React.DetailedHTMLProps<
    AnchorHTMLAttributes<HTMLAnchorElement>,
    HTMLAnchorElement
  >) => {
    if (!href) return <>{children}</>;
    return (
      <Link href={href} target="_blank" className="text-blue-500">
        {children}
      </Link>
    );
  },
};
```

新增自訂`a`標籤使用`nextjs`的`Link`並新增樣式以及開啟新分頁。
